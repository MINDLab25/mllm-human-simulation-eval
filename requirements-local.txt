#   conda create -n pmsv-local python=3.11 -y
#   conda activate pmsv-local
#
#   # 1. PyTorch — use cu128 build (CUDA 12.8 < driver 12.9 = compatible,
#   #    and >= 2.6.0 so torch.int1 is available; the PyPI default installs

#   pip install torch torchvision --index-url https://download.pytorch.org/whl/cu128
#
#   # 2. gptqmodel 2.0.0 — released alongside the model; has native Qwen2.5-Omni
#   #    support and does NOT pull in mainline transformers 5.x
#   pip install "gptqmodel==2.0.0+cu121torch2.5" \
#       --extra-index-url https://huggingface.github.io/autogptq-index/whl/cu121/
#   # If the above wheel is unavailable, try the plain PyPI release:
#   #   pip install gptqmodel
#
#   # 3. Remaining deps (includes the Qwen2.5-Omni transformers preview fork)
#   pip install -r requirements-local.txt
#
#   # 4. Install the project itself
#   pip install -e .
#
